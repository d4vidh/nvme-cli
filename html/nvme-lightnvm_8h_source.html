301 static inline struct ppa_addr generic_to_dev_addr(
302  struct nvme_nvm_addr_format *ppaf, struct ppa_addr r)
303 {
304  struct ppa_addr l;
305 
306  l.ppa = ((__u64)r.g.blk) << ppaf->blk_offset;
307  l.ppa |= ((__u64)r.g.pg) << ppaf->pg_offset;
308  l.ppa |= ((__u64)r.g.sec) << ppaf->sect_offset;
309  l.ppa |= ((__u64)r.g.pl) << ppaf->pln_offset;
310  l.ppa |= ((__u64)r.g.lun) << ppaf->lun_offset;
311  l.ppa |= ((__u64)r.g.ch) << ppaf->ch_offset;
312 
313  return l;
314 }
